int melody[]={262, 294, 330,350,392,440,494,523};

void setup()
{
  pinMode(2,INPUT);
  pinMode(8,OUTPUT);
}

void loop()
{
  int readValue=digitalRead(2);
  if(readValue==HIGH){
    digitalWrite(13,HIGH);
    for(int i=0;i<8;i++){
      tone(8,melody[i],300);
      delay(500);
    }
  }else{
    digitalWrite(13,LOW);
  }
}
